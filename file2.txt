import { Component } from '@angular/core';
import { QuestionnaireService } from './questionnaire.service';
import { Question } from '../models/question.model';

@Component({
  selector: 'app-questionnaire',
  templateUrl: './questionnaire.component.html',
  styleUrls: ['./questionnaire.component.css'],
})
export class QuestionnaireComponent {
  currentQuestion: Question | null = null;
  selectedAnswers: string[] = []; // Zmieniamy z pojedynczego wyboru na tablicę
  historyLength: number = 0;

  constructor(private questionnaireService: QuestionnaireService) {
    this.questionnaireService.currentQuestion$.subscribe((question) => {
      this.currentQuestion = question;
      this.selectedAnswers = []; // Czyścimy odpowiedzi przy nowym pytaniu
    });
  }

  toggleAnswer(answer: string) {
    const index = this.selectedAnswers.indexOf(answer);
    if (index === -1) {
      this.selectedAnswers.push(answer);
    } else {
      this.selectedAnswers.splice(index, 1);
    }
  }

  goNext() {
    if (this.selectedAnswers.length > 0) {
      this.questionnaireService.answerQuestion(this.selectedAnswers);
      this.historyLength++;
    }
  }

  goBack() {
    this.questionnaireService.goBack();
    this.historyLength--;
  }
}


<div class="container">
  <mat-card *ngIf="currentQuestion; else endTemplate" class="question-container">
    <mat-card-title class="question-title">{{ currentQuestion.text }}</mat-card-title>
    <mat-card-content>
      <mat-checkbox
        *ngFor="let answer of currentQuestion.answers"
        [checked]="selectedAnswers.includes(answer.text)"
        (change)="toggleAnswer(answer.text)"
      >
        {{ answer.text }}
      </mat-checkbox>
    </mat-card-content>

    <mat-card-actions class="button-group">
      <button mat-flat-button color="accent" (click)="goBack()" [disabled]="historyLength === 0">
        ⬅ Wstecz
      </button>
      <button mat-flat-button color="primary" (click)="goNext()" [disabled]="selectedAnswers.length === 0">
        Dalej ➡
      </button>
    </mat-card-actions>
  </mat-card>

  <ng-template #endTemplate>
    <mat-card class="question-container">
      <mat-card-title class="question-title">Dziękujemy za wypełnienie kwestionariusza!</mat-card-title>
      <mat-card-actions class="button-group">
        <button mat-flat-button color="primary" (click)="goBack()">⬅ Wróć do poprzedniego pytania</button>
      </mat-card-actions>
    </mat-card>
  </ng-template>
</div>
